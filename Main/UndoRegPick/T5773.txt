    LOCAL PROCEDURE CalcBaseQty@14(Qty@1000 : Decimal) : Decimal;
    BEGIN
      //>>C2-1.15
      IF "Qty. per Unit of Measure" = 0 THEN
        EXIT(Qty);
      EXIT(ROUND(Qty * "Qty. per Unit of Measure",0.00001));
      //<<C2-1.15
    END;

    PROCEDURE CalcMaxUndoQtyLine@1000000012() : Decimal;
    VAR
      ProdCompLine@1000000002 : Record 5407;
      MaxUndoQtyTmp@1000000003 : Decimal;
    BEGIN
      //>>C2-1.15
      CASE "Source Type" OF
        DATABASE::"Sales Line" :
          BEGIN
            MaxUndoQtyTmp := Quantity;
          END;
        DATABASE::"Prod. Order Component" :
          BEGIN
            IF ProdCompLine.GET("Source Subtype","Source No.","Source Line No.","Source Subline No.") THEN BEGIN
              MaxUndoQtyTmp := ProdCompLine."Qty. Picked" - (ProdCompLine."Expected Quantity" - ProdCompLine."Remaining Quantity");
              IF MaxUndoQtyTmp > Quantity THEN
                MaxUndoQtyTmp := Quantity;
            END;
          END;
      END;
      EXIT(MaxUndoQtyTmp);
      //<<C2-1.15
    END;

    PROCEDURE CalcQtyNotUndone@1000000001() : Decimal;
    VAR
      RegWhseActivityHeader@1000000003 : Record 5772;
      WarehouseEntry@1000000004 : Record 7312;
      TotalQty@1000000002 : Decimal;
      TotalQtyBase@1000000001 : Decimal;
    BEGIN
      WarehouseEntry.SETCURRENTKEY("Reference No.","Registering Date");
      WarehouseEntry.SETRANGE("Reference No.","No.");
      IF RegWhseActivityHeader.GET("Activity Type","No.") THEN
        WarehouseEntry.SETRANGE("Registering Date",RegWhseActivityHeader."Registering Date");
      CASE "Activity Type" OF
        "Activity Type"::"Put-away":
          WarehouseEntry.SETRANGE(
            "Reference Document",WarehouseEntry."Reference Document"::"Put-away");
        "Activity Type"::Pick:
          WarehouseEntry.SETRANGE(
            "Reference Document",WarehouseEntry."Reference Document"::Pick);
        "Activity Type"::Movement:
          WarehouseEntry.SETRANGE(
            "Reference Document",WarehouseEntry."Reference Document"::Movement);
      END;
      WarehouseEntry.SETRANGE("Location Code","Location Code");
      WarehouseEntry.SETRANGE("Zone Code","Zone Code");
      WarehouseEntry.SETRANGE("Bin Code","Bin Code");

      WarehouseEntry.SETRANGE("Whse. Document No.","Whse. Document No.");
      WarehouseEntry.SETRANGE("Whse. Document Type","Whse. Document Type");
      WarehouseEntry.SETRANGE("Whse. Document Line No.","Whse. Document Line No.");

      IF WarehouseEntry.FINDSET THEN BEGIN
        REPEAT
          TotalQty += WarehouseEntry.Quantity;
          TotalQtyBase += WarehouseEntry."Qty. (Base)";
        UNTIL WarehouseEntry.NEXT = 0;
        EXIT(TotalQty);
      END;
      EXIT(Quantity);
    END;

    BEGIN
